00040 string urlencode(const string &c)
00041 {
00042   string escaped="";
00043   int max = c.length();
00044   for(int i=0; i<max; i++)
00045     {
00046       if ( (48 <= c[i] && c[i] <= 57) ||//0-9
00047            (65 <= c[i] && c[i] <= 90) ||//abc...xyz
00048            (97 <= c[i] && c[i] <= 122) || //ABC...XYZ
00049            (c[i]=='~' || c[i]=='!' || c[i]=='*' || c[i]=='(' || c[i]==')' || c[i]=='\'')
00050            )
00051         {
00052           escaped.append( &c[i], 1);
00053         }
00054       else
00055         {
00056           escaped.append("%");
00057           escaped.append( char2hex(c[i]) );//converts char 255 to string "ff"
00058         }
00059     }
00060   return escaped;
