00026   // for 99% of cases.
00027   static std::string URL(std::string res)
00028   {
00029     for(int i=0; i<res.size(); i++)
00030       {
00031         switch(res[i])
00032           {
00033           case ' ': res.replace(i, 1, "%20"); break;
00034           case '!': res.replace(i, 1, "%21"); break;
00035           case '"': res.replace(i, 1, "%22"); break;
00036           case '\'': res.replace(i, 1, "%27"); break;
00037           case '(': res.replace(i, 1, "%28"); break;
00038           case ')': res.replace(i, 1, "%29"); break;
00039           case '+': res.replace(i, 1, "%2B"); break;
00040           }
00041       }
00042     return res;
00043   }
