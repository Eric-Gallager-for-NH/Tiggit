00048 struct RateSort : SortBase
00049 {
00050   RateSort(DataList &d) : SortBase(d) {}
00051   bool isLess(DataList::Entry &a, DataList::Entry &b)
00052   {
00053     if(a.rateCount == 0) a.rating = -1;
00054     if(b.rateCount == 0) b.rating = -1;
00055 
00056     // Does 'a' have a rating? If so, sort by rating. This also
00057     // covers the case where 'b' has no rating (b.rating = -1).
00058     if(a.rating >= 0)
00059       {
00060         // Sort equal ratings by name
00061         if(a.rating == b.rating)
00062           return TitleSort::isLessStatic(a, b);
00063         else
00064           return a.rating > b.rating;
00065       }
00066 
00067     // 'a' has no rating. What about b?
00068     else if(b.rating >= 0)
00069       // If yes, then always put 'b' first.
00070       return false;
00071 
00072     // Neither has a rating. Sort by title instead.
00073     else return TitleSort::isLessStatic(a, b);
00074   }
